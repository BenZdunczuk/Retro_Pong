46void accel_calibrate(int16_t offset_cb[3], uint16_t samples){

78int16_t accel_lpf_int(int16_t in, int16_t *prev, uint8_t alpha){
79 // y[n] = y[n-1] + alpha*(x[n] - y[n-1]) ; alpha w [0..256]
80 int32_t diff = (int32_t)in - *prev;
81 int32_t delta = (diff * alpha) >> 8; // podziel przez 256
82 *prev = (int16_t)((int32_t)*prev + delta);
83 return *prev;
84}
