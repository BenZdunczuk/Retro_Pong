129 /* gracz */
130 if (moveUp && posY > 0)
131 posY -= 6;
132 if (moveDown && posY < height())
133 posY += 6;
134 if(rotateLeft)
135 --rotationAnglePaddle;
136 if(rotateRight)
137 ++rotationAnglePaddle;
138
139 /* ai */
140 if (aiPaddle.center().y() < ball.center().y() && aiPaddle.bottom() < height())
141 aiPaddle.moveTop(aiPaddle.top() + 4);
142 else if (aiPaddle.center().y() > ball.center().y() && aiPaddle.top() > 0)
143 aiPaddle.moveTop(aiPaddle.top() - 4);
144
145 /* ruch piłki */
146 ball.translate(ballDX, ballDY);
147
148 /* odbicia */
149 if (ball.top() <= 0 || ball.bottom() >= height())
150 rotationAngleBall = rotation(reflaction(rotationAngleBall,180));
151
152 if (ball.intersects(playerPaddle) && ballDX > 0)
153 rotationAngleBall = rotation(reflaction(rotationAngleBall,rotationAnglePaddle + 90));
154 if (ball.intersects(aiPaddle) && ballDX < 0)
155 rotationAngleBall = rotation(reflaction(rotationAngleBall,90));
156
157 /* piłeczka wypada poza boisko */
158 if (ball.left() < 0){ //wypada ze strony AI
159 ++scorePlayer;
160 resetBall();
161 }
162
163 if (ball.right() > width()){ //wypada ze strony gracza
164 ++scoreAI;
165 resetBall();
166 }
