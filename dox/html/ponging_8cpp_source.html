241void PongWidget::calculateMovement(QStringList data){
242 //próg dla wartości sensora, po których jest odczytywany ruch liniowy i obrotowy paletki
243 int thresholdLinear = 2000;
244 int thresholdAngular = 4000;
245
246 //minimalne i maksymalne wartości możliwego ruchu
247 int minStepLinear = 5;
248 int maxStepLinear = 20;
249 int minStepAngular = 1;
250 int maxStepAngular = 5;
251
252 //zmiana pozycji(piksele) i kąta(stopnie) paletki
253 int da = 0;
254 int dy = 0;
255
256 //interpretacja danych z osi Y -- ruch paletki
257 if (data[1].toInt() < -thresholdLinear) {
258 dy = (((-data[1].toInt() - thresholdLinear) / 1000) + minStepLinear);
259 if (dy > maxStepLinear) dy = maxStepLinear;
260 } else if (data[1].toInt() > thresholdLinear) {
261 dy = -(((data[1].toInt() - thresholdLinear) / 1000) + minStepLinear);
262 if (dy < -maxStepLinear) dy = -maxStepLinear;
263 }
264 if(posY > 0 && posY < height()) posY += dy;
265 else if(posY <= 0) posY = 1;
266 else if(posY >= height()) posY = height()-1;
267
268 //interpretacja danych z osi X -- obrót paletki
269 if (data[0].toInt() < -thresholdAngular) {
270 da = (((-data[0].toInt() - thresholdAngular) / 1000) + minStepAngular);
271 if (da > maxStepAngular) da = maxStepAngular;
272 } else if (data[0].toInt() > thresholdAngular) {
273 da = -(((data[0].toInt() - thresholdAngular) / 1000) + minStepAngular);
274 if (da < -maxStepAngular) da = -maxStepAngular;
275 }
276 if(posY > 0 && posY < height()) rotationAnglePaddle += da;
277 rotationAnglePaddle %= 360;
278}
